import re\n\ndef extract_function(Column_Name, Excel_File):\n    results = []\n    confidence_score = 85\n\n    if not isinstance(Column_Name, str) or not Column_Name.strip():\n        results.append({"extractedValue": null, "validationStatus": "invalid", "aiReasoning": "Column_Name parameter is empty or not a string.", "confidenceScore": 0, "documentSource": "input"})\n        return results\n    if not isinstance(Excel_File, str) or not Excel_File.strip():\n        results.append({"extractedValue": null, "validationStatus": "invalid", "aiReasoning": "Excel_File parameter is empty or not a string.
