import openpyxl\nimport os\n\ndef extract_function(Column_Name, Excel_File):\n    """\n    Identifies the worksheet name for each specified column in an Excel file.\n\n    Args:\n        Column_Name (list): A list of column names (strings) to search for.\n        Excel_File (str): The path to the Excel file.\n\n    Returns:\n        dict: A dictionary where keys are the input column names and values are\n              the names of the worksheets where they were found. If a column\n              is not found in any worksheet, its value will be None.\n              Returns None if there's an error opening or reading the Excel file,\n              or if Column_Name is not a list.\n    """\n    if not isinstance(Column_Name, list):\n        print(\"Error: Column_Name must be a list of strings.\")\n        return None\n\n    if not os.path.exists(Excel_File):\n        print(f\"Error: Excel file not found at '{Excel_File}'\")\n        return None\n\n    if not os.path.isfile(Excel_File):\n        print(f\"Error: '{Excel_File}' is not a valid file.\")\n        return None\n\n    results = {}\n    for col in Column_Name:\n        results[col] = None # Initialize with None\n\n    try:\n        workbook = openpyxl.load_workbook(Excel_File)\n    except Exception as e:\n        print(f\"Error opening Excel file '{Excel_File}': {e}\")\n        return None\n\n    found_columns = set() # To track columns already found and avoid redundant searches\n\n    for sheet_name in workbook.sheetnames:\n        worksheet = workbook[sheet_name]\n        # Assuming headers are in the first row\n        header_row = [cell.value for cell in worksheet[1] if cell.value is not None]\n\n        for col_name_to_find in Column_Name:\n            if col_name_to_find in found_columns:\n                continue # Already found this column\n\n            if col_name_to_find in header_row:\n                results[col_name_to_find] = sheet_name\n                found_columns.add(col_name_to_find)\n\n    return results\n
